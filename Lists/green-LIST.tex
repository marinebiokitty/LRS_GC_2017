\DECLARESUBTYPE{Green}{Element}
\PRESETS{Green}{
  \FD\MYfile	{} %% filename (in \greens dir)
  \FS\MYtext	{\ifx\MYfile\empty\else%
		  \getextractenvs{document}{\greens/\MYfile}%
		\fi}
  }


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\NEW{Green}{\gDefendCities}{
  \s\MYname	{Defending Cities}
  \s\MYfile	{DefendCities.tex}
  }
\NEW{Green}{\gAttackCities}{
  \s\MYname	{Attacking Cities}
  \s\MYfile	{AttackCities.tex}
  }
