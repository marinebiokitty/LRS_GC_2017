%%%%%
%%
%% This file sets up the Abil datatype and creates Abil macros.  These
%% are for abilities that characters may have.
%%
%%%%%

\DECLARESUBTYPE{Abil}{Element}
\PRESETS{Abil}{
  \F\MYtext	%% text of ability, read by user
  \F\MYeffect	%% effect text of ability, read by recipient(s)
  }


%%%%%
%% \ability{<name>}{<text>}{<effect>}
%%
%% \ability is a wrapper around \INSTANCE, useful for 1-shot abilities,
%% etc.
\newinstance{Abil}{\ability[3]}{
  \s\MYname{#1}\s\MYtext{#2}\s\MYeffect{#3}}

\NEW{Abil}{\aMonologue}{
  \s\MYname	{Monologue}
  \s\MYeffect	{Everyone in the room must listen to my villainous monologue for thirty seconds.  They can continue working on tasks that won't interfere with listening.  I cannot be interrupted.}
  \s\MYtext	{Begin a villainous monologue with an evil laugh.  After this, you may speak for up to thirty seconds.  Everyone in the room when you begin your monologue must stay in the room and listen to you without interruptions.  
You may continue speaking after thirty seconds, but other characters may leave or interrupt at that time.
Other characters can continue tasks they were already working on as they listen, but cannot use other abilities.
Once you have finished one monologue, you cannot start another for five minutes.
\\ {\bf Usable twice.}
} % If they're picking a lock etc they can continue.
  }
\NEW{Abil}{\aMoralStand}{
  \s\MYname	{Moral Stand}
  \s\MYeffect	{I make a heroic speech.  Everyone in the room must listen to my heroic speech for thirty seconds.  They can continue working on tasks that won't interfere with listening.  I cannot be interrupted.}
  \s\MYtext	{Place your hand on your heart and look heroically into the distance.  Speak for up to thirty seconds on some matter of moral principle, such as loyalty, family unity, or honoring one's parents. Everyone in the room must stay in the room and listen to you.  You cannot be interrupted while you are speaking.
You may continue speaking after thirty seconds, but other characters may leave or interrupt at that time.
Other characters can continue tasks they were already working on as they listen, but cannot use other abilities. 
Once you have finished one speech, you cannot start another for five minutes.
\\ {\bf Usable twice.}
} % If they're picking a lock etc they can continue.

  }
\NEW{Abil}{\aFreezePerson}{
  \s\MYname	{Freeze Person}
  \s\MYeffect	{You are frozen in place for ninety seconds, unable to move, act, communicate, or be moved.  Other characters see you as covered in ice.}
  \s\MYtext	{The targeted character is frozen in place for two minutes, unable to move, act, communicate, or be moved.  Other characters can see you freezing the character, and see the target frozen in ice for the duration of the effect.
		 Once used, this ability cannot be used again for ten minutes. \\ {\bf Usable twice.  This is a superpower.}}
}
\NEW{Abil}{\aFreezeObject}{
  \s\MYname	{Freeze Object}
  \s\MYeffect   {I cover an object in ice.}
  \s\MYtext	{Place a purple sticker on an object, and write the current time plus five minutes on the sticker.  Other characters see the item as covered in ice until that time.  The item cannot be moved or used in that time, though characters can look at the items through the ice.  Other characters can see you freezing the object.
Once used, this ability cannot be used again for ten minutes.  \\ {\bf Usable twice.  This is a superpower.}}
  }
\NEW{Abil}{\aZapObject}{
  \s\MYname	{Zap Device}
  \s\MYeffect	{I temporarily short out an electronic device.}
  \s\MYtext	{Touch a device marked as tinkerable.  Mark it with a yellow sticker, with the current time plus five minutes.  That device cannot be used or tinkered with until that time, though it can be taken and moved.  Other characters can see you zapping the object. \\ {\bf Usable twice. This is a superpower.}
}
  }
\NEW{Abil}{\aZapPerson}{
  \s\MYname	{Zap Person}
  \s\MYeffect	{I give your body an electric charge.  You cannot use electronics, including phones, for three minutes.  You are aware of the effect, though it is not visible to others.}
  \s\MYtext	{If another person is in your ZoC, point to them and say "I zap you."  That person cannot use electronic devices, including phones, for three minutes.  You might use this, for example, to prevent someone from attacking a city.  The victim is aware of the effect, but others see nothing.
Once used, this ability cannot be used again for ten minutes. \\ {\bf Usable twice.  This is a superpower.}
}
  }
\NEW{Abil}{\aPhase}{
  \s\MYname	{Phase Through Walls}
  \s\MYeffect	{You see me walk through a wall or locked door.}
  \s\MYtext	{Walk to the other side, unless you are affected by an ability that prevents you from leaving the current room or conversation.  Characters on either side of the wall or door see this happen.   {\bf Usable at any time, as many times as you like.  This is a superpower.}}
  }
\NEW{Abil}{\aInvisibility}{
  \s\MYname	{Invisibility}
  \s\MYeffect	{I go invisible, though I can still be heard.}
  \s\MYtext	{Don a blue headband which marks you as invisible.  Other characters cannot see you, though they can hear you and recognize your voice if you speak.  You can return to visibility any time you like, but cannot stay invisible for longer than five minutes at a time.   Once you become visible again, this ability cannot be used for another minute. \\ {\bf Usable up to five times.  This is a superpower.}}
  }
\NEW{Abil}{\aSpeed}{
  \s\MYname	{Super Speed}
  \s\MYeffect	{I move to you at super speed, and see in detail what you were doing when I triggered this ability.}
  \s\MYtext	{If you see another character in the same room as you, you may point at them and say ``Zoom!'' loudly.  Move right next to them.  They must describe what they were doing when you said ``Zoom!''  You may speak to them, try to stop them, or otherwise interact with them as usual.  Other characters see you move. \\ {\bf Usable up to five times.  This is a superpower.}}
  }
\NEW{Abil}{\aHearing}{
  \s\MYname	{Super Hearing}
  \s\MYeffect	{I overhear your conversation from elsewhere in the room.  I seem to be doing whatever I was doing before; you cannot tell I am listening.}
  \s\MYtext	{If a conversation is taking place in the same room as you, you can choose to overhear it.  As a player, do one of the OOC gestures (e.g. crossed fingers, hand on your head, etc), walk over to the conversation, and listen in.  Other characters see you doing whatever you were doing before; e.g. if you were on your phone across the room, you still appear to be on your phone across the room; let anyone in the room know that's what they see.  This ability ends when the characters you were listening to finish their conversation or move out of the room, or if you leave the room.  You can only listen to one conversation at once.  {\bf Usable five times.  This is a superpower.}}
  }
\NEW{Abil}{\aNullification}{
  \s\MYname	{Nullification}
  \s\MYeffect	{You cannot use your superpower for three minutes after I touch you, up to a maximum of ten minutes.}
  \s\MYtext	{This ability {\emph automatically} triggers when you touch another character, whether you want it to trigger or not.  \\
That character's superpowers are nullified.  For the next three minutes, that character cannot use any ability which says ``This is a superpower.''  Repeated touches reset the three-minute timer, up to a maximum of ten minutes.  After the character regains their powers, their powers cannot be nullified again for ten minutes.  \\ {\bf This is a superpower, but it cannot be affected by other superpowers.}}
  }
\NEW{Abil}{\aAmplification}{
  \s\MYname	{Amplification}
  \s\MYeffect   {I enhance your superpowers.   Any of your abilities that say ``This is a superpower,'' that has effects lasting for $N$ minutes, lasts for $N+2$ minutes if used within five minutes of this ability.}
  \s\MYtext	{If another character is in touch range, you may touch them and amplify their superpowers.  For the next five minutes, that character's abilities are enhanced (see reverse for details). After you use this ability, you cannot use it again for ten minutes. \\ {\bf You've been holding off on using your power for a long time.  If you do not use this power within the first two hours of game, and the supercharged character does not then use their power, you will become wounded for 30 minutes.} \\ {\bf This is a superpower.}}
  }
\NEW{Abil}{\aIrresistible}{
  \s\MYname	{Irresistible}
  \s\MYeffect	{You find me irresistibly attractive.  You are compelled to stop what you are doing and converse with me for one minute.}
  \s\MYtext	{Your partner in a preexisting sexual relationship finds you irresistibly fascinating.  As far as they are concerned, you are the most interesting thing in the room for one minute.  They must stop what they are doing and converse with you during that time.  During that time, that character may only use touch-based abilities, and only on you.  Once this ability has been used, you may not use it again for twenty minutes.  \\ {\bf Usable twice.}}
  }

If the character is a member of the \cHeroLeague{\intro}, the PR score goes up by 1.  If the character is a member of the \cVillainCompact{\intro}, the PR score goes down by 1.
\\ {\bf If the character endorses you, this ability is consumed.}  If they do not, you may try this ability again.  You may not use this ability on the same character more than once every ten minutes.}
}
